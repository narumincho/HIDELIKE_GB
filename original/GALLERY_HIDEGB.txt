@FONTINIT
  VAR CHRCODE%,PAT$'OPTION STRICTように かってにつけたし
  RESTORE @FONTDATA
  WHILE 1
    READ CHRCODE%,PAT$
    IF CHRCODE%<0 THEN RETURN
    IF LEN(PAT$)==256 THEN
      ' 256 もじなら そのまま FONTDEF に わたす
      FONTDEF CHRCODE%,PAT$
    ELSE
      ' 16 もじの はずなので どくじかんすうに わたす
      SETFONT CHRCODE%,PAT$
    ENDIF
  WEND
  RETURN
' -------------------------------------------
' このまま つかえる フォントライブラリ

' みじかいデータもじれつ(4ドットあたり 1もじ)を
' していもじコード(CHRCODE%)に FONTDEF する
DEF SETFONT CHRCODE%,FONTPAT$
  VAR FONTI%
  VAR FONTJ%'OPTION STRICT ように かってに つけたし
  DIM FONTPAT%[8,8]
  FOR FONTI%=0 TO 7
    VAR FONTBYTE%=VAL("&H"+MID$(FONTPAT$,FONTI%*2,2))
    VAR BMASK%=128
    FOR FONTJ%=0 TO 7
      IF FONTBYTE% AND BMASK% THEN
        FONTPAT%[FONTJ%,FONTI%]=1
        'FONTPAT$=FONTPAT$+RGBA2HEX16(15,15,15,0)
      ENDIF
      BMASK%=BMASK%/2
    NEXT
  NEXT
  SETFONTPAT CHRCODE%,FONTPAT%
END

' 8X8はいれつ(FONTPAT%)に 0か1をいれた パターンを
' していもじコード(CHRCODE%)に FONTDEF する
DEF SETFONTPAT CHRCODE%,FONTPAT%
  VAR FONTI%,FONTJ%,FONTPAT$
  FOR FONTI%=0 TO 7
    FOR FONTJ%=0 TO 7
      IF FONTPAT%[FONTJ%,FONTI%] THEN
        ' コメントアウトを かえると バリエーションが かわります
        ' たて グラデーション
        'VAR FONTGRAD%=31-FONTI%*3
        'VAR FONTGRAD%=31-VAL("&H"+MID$("FC828CF",FONTI%,1))
        ' ななめ グラデーション
        'VAR FONTGRAD%=31-ABS(FONTI%-FONTJ%)*3
        'VAR FONTGRAD%=31-VAL("&H"+MID$("048BFB9",ABS((7-FONTI%)-FONTJ%),1))
        ' フラット
        VAR FONTGRAD%=31
        FONTPAT$=FONTPAT$+FONTRGBA2PAT(FONTGRAD%,FONTGRAD%,FONTGRAD%,1)
      ELSE
        ' かげを つける
        VAR FONTSHADOW%=0
        'IF FONTI%>0 AND FONTJ%>0 THEN FONTSHADOW%=FONTSHADOW% OR (FONTPAT%[FONTJ%-1,FONTI%-1]>0)
        'IF FONTI%>0 THEN FONTSHADOW%=FONTSHADOW% OR (FONTPAT%[FONTJ%,FONTI%-1]>0)
        'IF FONTJ%>0 THEN FONTSHADOW%=FONTSHADOW% OR (FONTPAT%[FONTJ%-1,FONTI%]>0)
        FONTPAT$=FONTPAT$+FONTRGBA2PAT(7,7,7,FONTSHADOW%)
      ENDIF
    NEXT
  NEXT
  FONTDEF CHRCODE%,FONTPAT$
END

' R,G,B(0~31) と A(0か1) を FONTDEF の 1ドットに へんかん
COMMON DEF FONTRGBA2PAT(R,G,B,A)
  RETURN RIGHT$("000"+HEX$(R*&H800+B*&H40+G*&H2+A),4)
END
